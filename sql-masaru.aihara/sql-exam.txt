問題7
SELECT pr.name,
CASE WHEN po.gender_code='m' THEN '男' 
WHEN po.gender_code='f' THEN '女' 
ELSE po.gender_code END AS gender,
SUM(po.population) as population 
FROM prefectures AS pr 
INNER JOIN populations AS po ON pr.code=po.prefecture_code
WHERE pr.region_code='80'
GROUP BY pr.code, pr.name, po.gender_code
ORDER BY pr.code ASC, po.gender_code DESC;
